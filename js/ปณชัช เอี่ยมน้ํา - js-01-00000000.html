03 จงเขียนโปรแกรม เพื่อรับคะแนน 3 ช่อง โดยละช่อง มีคะแนน 0-50 โดยคะแนนประกอบด้วย การบ้าน (hw) สอบกลางภาค (mid) สอบปลายภาค (fin) โดยคิดสัดส่วน ดังนี้
score = (hw(30%) + mid(30%) + fin(40%)) * 2
หลังจากนั้น ตัดเกรด 3 ระดับ ดังนี้

  • X - excellence หากคะแนนรวม มากกว่าเท่ากับ 85%
  • P - Pass หากคะแนนรวม มากกว่าเท่ากับ 60%
  • F - Fail หากคะแนนรวม น้อยกว่า 60%

HW Midterm Final
Score Grade

Answer: 

 
                function btnGrade_click(){
                    var hw = document.getElementById("iHW").value;
                    var mid = document.getElementById("iMid").value;
                    var fin = document.getElementById("iFin").value;
                    var showscore = document.getElementById("txtScore");
                    var showgrade = document.getElementById("txtGd");
                    score = ((hw * 0.3) + (mid * 0.3) + (fin * 0.4)) * 2;
                    if (score > 100){
                        showscore.textContent = "Null";
                        showgrade.textContent = "Null";
                    }else if (score >= 85){
                        showscore.textContent = score;
                        showgrade.textContent = "X";
                    }else if (score >= 60){
                        showscore.textContent = score;
                        showgrade.textContent = "P";
                    }else{
                        showscore.textContent = score;
                        showgrade.textContent = "F";
                    }
                }

                function btnClick(obj){
                    //alert(obj.id)
                    var inpa = Number(document.getElementById("inpA").value)
                    var inpb = Number(document.getElementById("inpB").value)
                    var textresult = document.getElementById("txtResult")

                    }
