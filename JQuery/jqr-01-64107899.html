03 จงเขียนโปรแกรม เพื่อรับคะแนน 3 ช่อง โดยละช่อง มีคะแนน 0-50 โดยคะแนนประกอบด้วย การบ้าน (hw) สอบกลางภาค (mid) สอบปลายภาค (fin) โดยคิดสัดส่วน ดังนี้
score = (hw(30%) + mid(30%) + fin(40%)) * 2
หลังจากนั้น ตัดเกรด 3 ระดับ ดังนี้

  • X - excellence หากคะแนนรวม มากกว่าเท่ากับ 85%
  • P - Pass หากคะแนนรวม มากกว่าเท่ากับ 60%
  • F - Fail หากคะแนนรวม น้อยกว่า 60%

HW Midterm Final
Score Grade

Answer: 

 
                $(() =>{
                    $('#btnGrade').click(btnGrade_click);
                   })
    
                   function btnGrade_click(){
                    var hw = $("#iHW").val();
                    var mid = $("#iMid").val();
                    var fin = $("#iFin").val();
                    var score = ((hw * 0.3) + (mid * 0.3) + (fin * 0.4)) * 2;
                    if (score > 100){
                        g = "NOT"
                    }else if (score >= 85){
                        g = "X"
                    }else if (score >= 60){
                        g = "p"
                    }else{
                        g = "F"
                    }   
                    $("#txtScore").text(score);
                    $("#txtGd").text(g);
                   }

                    function btnClick(obj) {
                        alert(obj.id)   
                        var inpa = parseFloat($('#inpA').val());
                        var inpb = parseFloat($('#inpB').val());

        
                        if (obj.id == 'btnAdd') {
                            r = inpa + inpb
                        } else if (obj.id == 'btnSub') {
                            r = inpa - inpb
                        } else if (obj.id == 'btnMul') {
                            r = inpa * inpb
                        } else if (obj.id == 'btnDiv') {
                            r = inpa / inpb
                        }
        
                        $('#txtResult').text(r);
                }
